-- ============================================================
-- SECTION: Unload (Export) Data to GCS
-- ============================================================
-- LEARNING NOTES:
--   Shows how to export/unload data FROM a Snowflake table TO a cloud stage
--   using COPY INTO @stage syntax (reverse of the normal load direction).
-- KEY INTERVIEW CONCEPTS:
--   - COPY INTO @stage_name FROM table_name exports data to cloud storage
--   - Unloading writes files (CSV, Parquet, JSON) to the stage location
--   - ALTER STORAGE INTEGRATION can update allowed locations
-- REMEMBER:
--   - Loading = COPY INTO table FROM @stage; Unloading = COPY INTO @stage FROM table
--   - Unloaded files are split into multiple parts by default for parallelism
-- ============================================================

------- Unload data -----
USE ROLE ACCOUNTADMIN;
USE DATABASE DEMO_DB;


-- create integration object that contains the access information
CREATE STORAGE INTEGRATION gcp_integration
  TYPE = EXTERNAL_STAGE
  STORAGE_PROVIDER = GCS
  ENABLED = TRUE
  STORAGE_ALLOWED_LOCATIONS = ('gcs://snowflakebucketgcp', 'gcs://snowflakebucketgcpjson');
  
  
-- create file format
create or replace file format demo_db.public.fileformat_gcp
    TYPE = CSV
    FIELD_DELIMITER = ','
    SKIP_HEADER = 1;

-- create stage object
create or replace stage demo_db.public.stage_gcp
    STORAGE_INTEGRATION = gcp_integration
    URL = 'gcs://snowflakebucketgcp/csv_happiness'
    FILE_FORMAT = fileformat_gcp
   
    ;


ALTER STORAGE INTEGRATION gcp_integration
SET  storage_allowed_locations=('gcs://snowflakebucketgcp', 'gcs://snowflakebucketgcpjson')

SELECT * FROM HAPPINESS;

COPY INTO @stage_gcp
FROM
HAPPINESS;
