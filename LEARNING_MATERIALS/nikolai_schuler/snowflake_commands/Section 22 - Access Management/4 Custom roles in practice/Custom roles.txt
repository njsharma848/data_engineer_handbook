-- ============================================================
-- SECTION 22.4: CUSTOM ROLES IN PRACTICE
-- ============================================================
-- LEARNING NOTES:
--   This section demonstrates how custom roles (e.g., SALES_ADMIN,
--   SALES_USERS) operate in practice -- creating tables, granting
--   granular privileges, and validating access controls.
-- KEY INTERVIEW CONCEPTS:
--   - Custom roles enable granular, least-privilege access control
--   - GRANT SELECT allows read-only access; GRANT DELETE allows row deletion
--   - Privilege grants are explicit: USAGE on database, USAGE on schema, SELECT on table
--   - Role admins (SALES_ADMIN) manage privileges for their department users
--   - Users can only perform actions allowed by their current role's privileges
--   - DROP TABLE requires OWNERSHIP privilege, not just DELETE
--   - Three-level grant pattern: DATABASE usage -> SCHEMA usage -> TABLE privileges
-- REMEMBER: In Snowflake RBAC, privileges are never implicit. You must
--   grant USAGE on the database AND schema before any table-level
--   privileges will work -- a common pitfall in interviews.
-- ============================================================

USE ROLE SALES_ADMIN;
USE SALES_DATABASE;

-- Create table --
create or replace table customers(
  id number,
  full_name varchar, 
  email varchar,
  phone varchar,
  spent number,
  create_date DATE DEFAULT CURRENT_DATE);

-- query from table --
SELECT* FROM CUSTOMERS;
USE ROLE SALES_USERS;

-- grant usage to role
USE ROLE SALES_ADMIN;

GRANT USAGE ON DATABASE SALES_DATABASE TO ROLE SALES_USERS;
GRANT USAGE ON SCHEMA SALES_DATABASE.PUBLIC TO ROLE SALES_USERS;
GRANT SELECT ON TABLE SALES_DATABASE.PUBLIC.CUSTOMERS TO ROLE SALES_USERS


-- Validate privileges --
USE ROLE SALES_USERS;
SELECT* FROM CUSTOMERS;
DROP TABLE CUSTOMERS;
DELETE FROM CUSTOMERS;
SHOW TABLES;

-- grant DROP on table
USE ROLE SALES_ADMIN;
GRANT DELETE ON TABLE SALES_DATABASE.PUBLIC.CUSTOMERS TO ROLE SALES_USERS
