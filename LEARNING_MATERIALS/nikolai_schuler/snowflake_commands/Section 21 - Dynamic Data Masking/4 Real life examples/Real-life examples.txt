-- ============================================================
-- SECTION 21.4: REAL-LIFE DYNAMIC DATA MASKING EXAMPLES
-- ============================================================
-- LEARNING NOTES:
--   This section provides practical masking policy examples
--   including partial email masking with regexp_replace, SHA-2
--   hashing for pseudonymization, and date masking strategies.
-- KEY INTERVIEW CONCEPTS:
--   - regexp_replace() enables partial masking (e.g., hide email username but keep domain)
--   - SHA2() hashing provides consistent pseudonymization (same input = same hash)
--   - date_from_parts() can mask dates by replacing with a sentinel value
--   - Different roles can see different levels of masking (full, partial, none)
--   - Masking policies can handle multiple data types (varchar, date, number)
--   - SHA2 masking preserves referential integrity for joins on masked data
--   - Partial masking is useful when some context must be preserved (e.g., email domain)
-- REMEMBER: For interviews, know three masking strategies -- full
--   replacement (static string), partial masking (regexp_replace),
--   and hashing (SHA2) for pseudonymization that preserves join capability.
-- ============================================================



### More examples - 1 - ###

USE ROLE ACCOUNTADMIN;

create or replace masking policy emails as (val varchar) returns varchar ->
case
  when current_role() in ('ANALYST_FULL') then val
  when current_role() in ('ANALYST_MASKED') then regexp_replace(val,'.+\@','*****@') -- leave email domain unmasked
  else '********'
end;


-- apply policy
ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN email
SET MASKING POLICY emails;


-- Validating policies
USE ROLE ANALYST_FULL;
SELECT * FROM CUSTOMERS;

USE ROLE ANALYST_MASKED;
SELECT * FROM CUSTOMERS;

USE ROLE ACCOUNTADMIN;


### More examples - 2 - ###


create or replace masking policy sha2 as (val varchar) returns varchar ->
case
  when current_role() in ('ANALYST_FULL') then val
  else sha2(val) -- return hash of the column value
end;



-- apply policy
ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN full_name
SET MASKING POLICY sha2;

ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN full_name
UNSET MASKING POLICY;


-- Validating policies
USE ROLE ANALYST_FULL;
SELECT * FROM CUSTOMERS;

USE ROLE ANALYST_MASKED;
SELECT * FROM CUSTOMERS;

USE ROLE ACCOUNTADMIN;


### More examples - 3 - ###

create or replace masking policy dates as (val date) returns date ->
case
  when current_role() in ('ANALYST_FULL') then val
  else date_from_parts(0001, 01, 01)::date -- returns 0001-01-01 00:00:00.000
end;


-- Apply policy on a specific column 
ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN create_date 
SET MASKING POLICY dates;


-- Validating policies

USE ROLE ANALYST_FULL;
SELECT * FROM CUSTOMERS;

USE ROLE ANALYST_MASKED;
SELECT * FROM CUSTOMERS;
