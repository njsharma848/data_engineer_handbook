-- ============================================================
-- SECTION 19.7: CHANGES CLAUSE (LIGHTWEIGHT CHANGE TRACKING)
-- ============================================================
-- LEARNING NOTES:
--   This section covers the CHANGES clause, an alternative to
--   streams for querying historical changes on a table using
--   Time Travel, without creating a persistent stream object.
-- KEY INTERVIEW CONCEPTS:
--   - CHANGE_TRACKING = TRUE must be enabled on the table
--     via ALTER TABLE before using the CHANGES clause.
--   - Syntax: SELECT * FROM table CHANGES(information => default)
--     AT (offset => -n) or AT (timestamp => '...').
--   - information => default returns all changes (like a
--     standard stream); information => append_only returns
--     only inserts.
--   - The AT clause uses Time Travel to specify the starting
--     point -- either by offset (seconds) or timestamp.
--   - Unlike streams, the CHANGES clause is stateless -- it
--     does not track an offset or get consumed.
--   - Useful for ad-hoc change auditing and one-time data
--     extraction without the overhead of maintaining a stream.
--   - Results can be materialized with CREATE TABLE ... AS
--     SELECT * FROM table CHANGES(...) AT (...).
-- REMEMBER: CHANGES clause = stream-like query without a
--   persistent object. Great for ad-hoc auditing, but use
--   streams for automated pipelines.
-- ============================================================

----- Change clause ------

--- Create example db & table ---

CREATE OR REPLACE DATABASE SALES_DB;

create or replace table sales_raw(
	id varchar,
	product varchar,
	price varchar,
	amount varchar,
	store_id varchar);

-- insert values
insert into sales_raw
	values
		(1, 'Eggs', 1.39, 1, 1),
		(2, 'Baking powder', 0.99, 1, 1),
		(3, 'Eggplants', 1.79, 1, 2),
		(4, 'Ice cream', 1.89, 1, 2),
		(5, 'Oats', 1.98, 2, 1);

ALTER TABLE sales_raw
SET CHANGE_TRACKING = TRUE;

SELECT * FROM SALES_RAW
CHANGES(information => default)
AT (offset => -0.5*60)


SELECT CURRENT_TIMESTAMP;

-- Insert values
INSERT INTO SALES_RAW VALUES (6, 'Bread', 2.99, 1, 2);
INSERT INTO SALES_RAW VALUES (7, 'Onions', 2.89, 1, 2);


SELECT * FROM SALES_RAW
CHANGES(information  => default)
AT (timestamp => 'your-timestamp'::timestamp_tz)

UPDATE SALES_RAW
SET PRODUCT = 'Toast2' WHERE ID=6;

CHANGES(information  => default)
AT (timestamp => 'your-timestamp'::timestamp_tz)


SELECT * FROM SALES_RAW
CHANGES(information  => append_only)
AT (timestamp => 'your-timestamp'::timestamp_tz)





CREATE OR REPLACE TABLE PRODUCTS
AS
SELECT * FROM SALES_RAW
CHANGES(information  => append_only)
AT (timestamp => 'your-timestamp'::timestamp_tz)


SELECT * FROM PRODUCTS;
