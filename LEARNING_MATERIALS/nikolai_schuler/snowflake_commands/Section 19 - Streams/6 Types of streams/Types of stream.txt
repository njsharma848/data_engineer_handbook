-- ============================================================
-- SECTION 19.6: TYPES OF STREAMS (DEFAULT VS APPEND-ONLY)
-- ============================================================
-- LEARNING NOTES:
--   This section compares the two main stream types in Snowflake:
--   the default (standard/delta) stream that captures all DML
--   changes, and the append-only stream that only tracks INSERTs.
-- KEY INTERVIEW CONCEPTS:
--   - Default (standard) streams capture INSERT, UPDATE, and
--     DELETE operations -- full CDC capability.
--   - Append-only streams (APPEND_ONLY = TRUE) only capture
--     INSERT operations; UPDATEs and DELETEs are ignored.
--   - Append-only streams are ideal for event/log tables where
--     data is only ever inserted, never modified or deleted.
--   - Consuming a stream can be done via DML (MERGE, INSERT)
--     or even CREATE TABLE ... AS SELECT * FROM stream.
--   - Default streams show UPDATE as DELETE + INSERT pairs;
--     append-only streams skip these entirely.
--   - Choose stream type based on your source table's DML
--     patterns and downstream processing needs.
--   - There is also an insert-only stream type for external
--     tables (not covered here).
-- REMEMBER: Use APPEND_ONLY = TRUE for insert-heavy tables
--   (logs, events) to simplify processing and reduce overhead.
-- ============================================================

------- Append-only type ------
USE STREAMS_DB;
SHOW STREAMS;



SELECT * FROM SALES_RAW_STAGING;

-- Create stream with default
CREATE OR REPLACE STREAM SALES_STREAM_DEFAULT
ON TABLE SALES_RAW_STAGING;

-- Create stream with append-only
CREATE OR REPLACE STREAM SALES_STREAM_APPEND
ON TABLE SALES_RAW_STAGING
APPEND_ONLY = TRUE;

-- View streams
SHOW STREAMS;


-- Insert values
INSERT INTO SALES_RAW_STAGING VALUES (14,'Honey',4.99,1,1);
INSERT INTO SALES_RAW_STAGING VALUES (15,'Coffee',4.89,1,2);
INSERT INTO SALES_RAW_STAGING VALUES (15,'Coffee',4.89,1,2);




SELECT * FROM SALES_STREAM_APPEND;
SELECT * FROM SALES_STREAM_DEFAULT;

-- Delete values
SELECT * FROM SALES_RAW_STAGING


DELETE FROM SALES_RAW_STAGING WHERE ID=7;


SELECT * FROM SALES_STREAM_APPEND;
SELECT * FROM SALES_STREAM_DEFAULT;


-- Consume stream via "CREATE TABLE ... AS"
CREATE OR REPLACE TEMPORARY TABLE PRODUCT_TABLE
AS SELECT * FROM SALES_STREAM_DEFAULT;
CREATE OR REPLACE TEMPORARY TABLE PRODUCT_TABLE
AS SELECT * FROM SALES_STREAM_APPEND;


-- Update
UPDATE SALES_RAW_STAGING
SET PRODUCT = 'Coffee 200g'
WHERE PRODUCT ='Coffee';


SELECT * FROM SALES_STREAM_APPEND;
SELECT * FROM SALES_STREAM;
