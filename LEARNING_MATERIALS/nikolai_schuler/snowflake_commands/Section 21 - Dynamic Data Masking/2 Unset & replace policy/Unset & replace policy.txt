-- ============================================================
-- SECTION 21.2: UNSETTING AND REPLACING MASKING POLICIES
-- ============================================================
-- LEARNING NOTES:
--   This section covers how to unset (remove) masking policies
--   from columns, replace existing policies with new ones, and
--   use metadata functions to inspect applied policies.
-- KEY INTERVIEW CONCEPTS:
--   - UNSET MASKING POLICY removes a policy from a column without dropping it
--   - You must UNSET a policy before you can DROP or REPLACE it if it is in use
--   - DESC MASKING POLICY shows the policy definition details
--   - SHOW MASKING POLICIES lists all masking policies in the current schema
--   - information_schema.policy_references() shows which columns use a policy
--   - A masking policy can be applied to multiple columns of the same data type
--   - Replacing a policy requires unsetting it from all columns first
-- REMEMBER: You cannot drop or replace a masking policy while it is
--   attached to any column -- always UNSET first, then DROP/REPLACE,
--   then re-apply the new policy.
-- ============================================================

 #### More examples  #####

USE ROLE ACCOUNTADMIN;

--- 1) Apply policy to multiple columns

-- Apply policy on a specific column 
ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN full_name 
SET MASKING POLICY phone;

-- Apply policy on another specific column 
ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN phone
SET MASKING POLICY phone;



--- 2) Replace or drop policy

DROP masking policy phone;

create or replace masking policy phone as (val varchar) returns varchar ->
            case
            when current_role() in ('ANALYST_FULL', 'ACCOUNTADMIN') then val
            else CONCAT(LEFT(val,2),'*******')
            end;

-- List and describe policies
DESC MASKING POLICY phone;
SHOW MASKING POLICIES;

-- Show columns with applied policies
SELECT * FROM table(information_schema.policy_references(policy_name=>'phone'));


-- Remove policy before replacing/dropping 
ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN full_name 
SET MASKING POLICY phone;

ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN email
UNSET MASKING POLICY;

ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN phone
UNSET MASKING POLICY;



-- replace policy
create or replace masking policy names as (val varchar) returns varchar ->
            case
            when current_role() in ('ANALYST_FULL', 'ACCOUNTADMIN') then val
            else CONCAT(LEFT(val,2),'*******')
            end;

-- apply policy
ALTER TABLE IF EXISTS CUSTOMERS MODIFY COLUMN full_name
SET MASKING POLICY names;


-- Validating policies
USE ROLE ANALYST_FULL;
SELECT * FROM CUSTOMERS;

USE ROLE ANALYST_MASKED;
SELECT * FROM CUSTOMERS;
