-- ============================================================
-- SECTION 22.2: SECURITYADMIN IN PRACTICE
-- ============================================================
-- LEARNING NOTES:
--   This section demonstrates how SECURITYADMIN creates custom
--   role hierarchies, assigns roles to users, and follows the
--   best practice of granting custom roles up to SYSADMIN.
-- KEY INTERVIEW CONCEPTS:
--   - SECURITYADMIN manages roles and users (CREATE ROLE, CREATE USER, GRANT ROLE)
--   - Role hierarchies are built by granting child roles to parent roles
--   - Best practice: always grant custom roles up to SYSADMIN for visibility
--   - If a role is NOT granted to SYSADMIN, SYSADMIN cannot manage its objects
--   - SECURITYADMIN inherits the USERADMIN role capabilities
--   - Role hierarchy example: sales_users -> sales_admin -> SYSADMIN
--   - Each functional area (Sales, HR) should have its own admin and user roles
-- REMEMBER: Always connect custom roles to SYSADMIN in the hierarchy.
--   If you skip this, SYSADMIN loses visibility into those roles'
--   objects -- a common interview gotcha about Snowflake RBAC.
-- ============================================================

-- SECURITYADMIN role --
--  Create and Manage Roles & Users --


-- Create Sales Roles & Users for SALES--

create role sales_admin;
create role sales_users;

-- Create hierarchy
grant role sales_users to role sales_admin;

-- As per best practice assign roles to SYSADMIN
grant role sales_admin to role SYSADMIN;


-- create sales user
CREATE USER simon_sales PASSWORD = '123' DEFAULT_ROLE =  sales_users 
MUST_CHANGE_PASSWORD = TRUE;
GRANT ROLE sales_users TO USER simon_sales;

-- create user for sales administration
CREATE USER olivia_sales_admin PASSWORD = '123' DEFAULT_ROLE =  sales_admin
MUST_CHANGE_PASSWORD = TRUE;
GRANT ROLE sales_admin TO USER  olivia_sales_admin;

-----------------------------------

-- Create Sales Roles & Users for HR--

create role hr_admin;
create role hr_users;

-- Create hierarchy
grant role hr_users to role hr_admin;

-- This time we will not assign roles to SYSADMIN (against best practice)
-- grant role hr_admin to role SYSADMIN;


-- create hr user
CREATE USER oliver_hr PASSWORD = '123' DEFAULT_ROLE =  hr_users 
MUST_CHANGE_PASSWORD = TRUE;
GRANT ROLE hr_users TO USER oliver_hr;

-- create user for sales administration
CREATE USER mike_hr_admin PASSWORD = '123' DEFAULT_ROLE =  hr_admin
MUST_CHANGE_PASSWORD = TRUE;
GRANT ROLE hr_admin TO USER mike_hr_admin;
