-- ============================================================
-- SECTION 6.39: HANDLING NESTED JSON DATA (JSON Part 3)
-- ============================================================
-- LEARNING NOTES:
--   This section covers accessing nested JSON objects with dot
--   notation, array indexing with [n], and UNION ALL for arrays.
-- KEY INTERVIEW CONCEPTS:
--   - Dot notation: RAW_FILE:job.salary for nested object access
--   - Array indexing: RAW_FILE:prev_company[0] for array elements
--   - ARRAY_SIZE() to count elements in a JSON array
--   - UNION ALL pattern to manually unnest array elements into rows
-- REMEMBER: Array indexes are 0-based. UNION ALL unnesting is
--   manual; use FLATTEN() (next section) for dynamic array sizes.
-- ============================================================

   // Handling nested data

SELECT RAW_FILE:job as job  FROM OUR_FIRST_DB.PUBLIC.JSON_RAW;


SELECT 
      RAW_FILE:job.salary::INT as salary
FROM OUR_FIRST_DB.PUBLIC.JSON_RAW;



SELECT 
    RAW_FILE:first_name::STRING as first_name,
    RAW_FILE:job.salary::INT as salary,
    RAW_FILE:job.title::STRING as title
FROM OUR_FIRST_DB.PUBLIC.JSON_RAW;


// Handling arrays

SELECT
    RAW_FILE:prev_company as prev_company
FROM OUR_FIRST_DB.PUBLIC.JSON_RAW;

SELECT
    RAW_FILE:prev_company[1]::STRING as prev_company
FROM OUR_FIRST_DB.PUBLIC.JSON_RAW;


SELECT
    ARRAY_SIZE(RAW_FILE:prev_company) as prev_company
FROM OUR_FIRST_DB.PUBLIC.JSON_RAW;




SELECT 
    RAW_FILE:id::int as id,  
    RAW_FILE:first_name::STRING as first_name,
    RAW_FILE:prev_company[0]::STRING as prev_company
FROM OUR_FIRST_DB.PUBLIC.JSON_RAW
UNION ALL 
SELECT 
    RAW_FILE:id::int as id,  
    RAW_FILE:first_name::STRING as first_name,
    RAW_FILE:prev_company[1]::STRING as prev_company
FROM OUR_FIRST_DB.PUBLIC.JSON_RAW
ORDER BY id
